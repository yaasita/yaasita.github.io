perlでDBMファイルを取り扱う

最近はtieという関数を使うらしい・・・


#!/usr/bin/perl
use feature qw(say);
use SDBM_File;
use Fcntl;
tie (%HA,'SDBM_File',"test",O_RDWR|O_CREATE,0644);
%HA=(one=>1,two=>2,three=>3);
while ( ($a,$b)=each %HA){
say "$a -> $b";
}
untie %HA;
